def best_response_value(history, br_player):
    if terminal(history): return payoff(history, br_player)
    if chance(history):
        return sum(p(a)*best_response_value(history+a, br_player) for a in actions)
    if current_player(history) == br_player:
        return max(best_response_value(history+a, br_player) for a in actions)
    else:  # opponent
        info = information_set(history)
        return sum(sigma[info][a] * best_response_value(history+a, br_player) for a in actions)
