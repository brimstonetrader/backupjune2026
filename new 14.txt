#include "types.h"
#include "stat.h"
#include "user.h"

#define NUM_CHILDREN 3

void
worker(int id, int tickets, int iterations)
{
  int i, j;
  int count = 0;
  int start_time = uptime();
  
  for(i = 0; i < iterations; i++) {
    // Do busy work to consume CPU time
    for(j = 0; j < 1000000; j++) {
      count++;
    }
  }
  
  int end_time = uptime();
  
  // Stagger output
  sleep(id);
  
  printf(1, "Process %d (%d tickets) completed in %d ticks\n", 
         getpid(), tickets, end_time - start_time);
}

int
main(int argc, char *argv[])
{
  int i;
  int pid;
  int ticket_counts[NUM_CHILDREN] = {10, 20, 70};  // 10%, 20%, 70% of tickets
  
  printf(1, "Starting lottery scheduler test...\n");
  printf(1, "Creating %d processes with different ticket counts\n", NUM_CHILDREN);
  printf(1, "Ticket distribution: 10, 20, 70 (10%%, 20%%, 70%%)\n\n");
  
  for(i = 0; i < NUM_CHILDREN; i++) {
    pid = fork();
    
    if(pid < 0) {
      printf(1, "Fork failed\n");
      exit();
    }
    
    if(pid == 0) {
      // Child process
      settickets(ticket_counts[i]);
      
      printf(1, "Child %d started with %d tickets\n", 
             getpid(), ticket_counts[i]);
      
      // Wait for all children to be created
      sleep(10);
      
      worker(i, ticket_counts[i], 20);
      exit();
    }
    
    sleep(1);
  }
  
  printf(1, "\nParent waiting for children to complete...\n\n");
  
  for(i = 0; i < NUM_CHILDREN; i++) {
    wait();
  }
  
  printf(1, "\nAll processes completed!\n");
  printf(1, "Expected: Process with 70 tickets should complete fastest\n");
  printf(1, "          Process with 10 tickets should complete slowest\n");
  
  exit();
}
