#include "types.h"
#include "stat.h"
#include "user.h"
#define NUM_CHILDREN 3

void
worker(int id, int iterations)
{
  int i, j;
  int count = 0;
  int start_time = uptime();
  
  for(i = 0; i < iterations; i++) {
    for(j = 0; j < 1000000; j++) { count++; } 
}
  
  int end_time = uptime();
  sleep(id);
  printf(1, "Process %d (age %d) completed in %d ticks\n", getpid(), id, end_time - start_time);
}

int
main(int argc, char *argv[])
{
  int i;
  int pid;
  int ages[NUM_CHILDREN] = {10, 5, 1};  
  
  printf(1, "Starting age scheduler test...\n");
  printf(1, "Creating %d processes with different age\n\n", NUM_CHILDREN);
  
  for(i = 0; i < NUM_CHILDREN; i++) {
    pid = fork();
    if(pid < 0) {
      printf(1, "Fork failed\n");
      exit();
    }
    if(pid == 0) {
      setage(ages[i]);
      printf(1, "Child %d started with age %d\n", getpid(), ages[i]);
      sleep(10);
      worker(ages[i], 50);
      exit();
    }
    sleep(1);
  }
  
  printf(1, "\nParent waiting for children to complete...\n\n");
  for(i = 0; i < NUM_CHILDREN; i++) { wait(); }
  printf(1, "\nAll processes completed!\n");
  exit();
}
