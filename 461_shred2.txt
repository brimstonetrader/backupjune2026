static void
read_block_range(char* dst, uint start_block, uint count) {
  for (int block=start_block;block<start_block+count;block++) {
      struct buf *bp = bread(2, block);
      memmove(dst,bp->data,BSIZE);
      brelse(bp);
      dst+=BSIZE;
  }
}

void
iso9660fs_ipopulate(struct inode* ip)
{
    uint addr = ip->inum;
    static char buf[1024];
    read_block_range(buf, addr/512, 2);

    struct iso9660_dir_s *entry =
        (struct iso9660_dir_s*)(buf + (addr & 0x1ff));

    ip->type  = (entry->file_flags & 0x2) ? T_DIR : T_FILE;
    ip->nlink = 1;
    ip->size  = entry->size;
    ip->addrs[0] = entry->extent * 2048;
    ip->flags |= I_VALID;
}

int
iso9660fs_readi(struct inode* ip, char* dst, uint offset, uint size)
{
  if ( ip->type == T_DIR ) {
    // Calculate which directory entry we're being asked for
    uint entry_index = offset / sizeof(struct dirent);
    
    // Handle . and .. entries first
    if (entry_index == 0) {
      // "." entry
      struct dirent *de = (struct dirent*)dst;
      memmove(de->name, ".", 2);
      de->inum = ip->inum;
      return sizeof(struct dirent);
    } else if (entry_index == 1) {
      // ".." entry  
      struct dirent *de = (struct dirent*)dst;
      memmove(de->name, "..", 3);
      de->inum = ip->inum; // For root, parent is itself
      return sizeof(struct dirent);
    }
    
    // For actual ISO directory entries, adjust index
    uint iso_entry_index = entry_index - 2;
    
    // Read directory data
    uint dir_extent = ip->addrs[0] / 512; // Convert byte address to block number
    uint dir_size = ip->size;
    
    // Read the directory into a buffer (we need to read enough blocks to cover the directory)
    uint blocks_to_read = (dir_size + BSIZE - 1) / BSIZE;
    if (blocks_to_read > 8) blocks_to_read = 8; // Limit for safety
    
    char dir_buf[8 * BSIZE]; // Buffer large enough for directory
    read_block_range(dir_buf, dir_extent, blocks_to_read);
    
    // Traverse directory entries to find the requested one
    struct iso9660_dir_s *entry = (struct iso9660_dir_s*)dir_buf;
    uint current_iso_index = 0;
    
    while ((char*)entry < dir_buf + dir_size && entry->length > 0) {
      // Skip the ISO's own . and .. entries
      if (entry->filename.len > 1 && 
          entry->filename.str[0] != 0 &&  // Not the current directory entry
          entry->filename.str[0] != 1) {  // Not the parent directory entry
        
        if (current_iso_index == iso_entry_index) {
          // Found the entry we're looking for
          struct dirent *de = (struct dirent*)dst;
          
          // Extract and clean the filename
          char clean_name[256];
          int name_len = entry->filename.len;
          int clean_len = name_len;
          
          memmove(clean_name, &entry->filename.str[1], name_len);
          clean_name[clean_len] = '\0';

          
          // Copy filename (skip the first byte which is file identifier length)
          if (clean_len >= 2 && 
              clean_name[clean_len-2] == ';' && 
              clean_name[clean_len-1] == '1') {
            clean_len -= 2;
            clean_name[clean_len] = '\0';
          }
          memmove(de->name, clean_name, clean_len + 1);
          
          // Create a pseudo-inode number from the entry's location
          // We'll use the extent as the high part and some identifier as low part
          de->inum = (entry->extent << 8) | (entry->filename.str[0] & 0xFF);
          
          return sizeof(struct dirent);
        }
        current_iso_index++;
      }
      
      // Move to next directory entry
      entry = (struct iso9660_dir_s*)((char*)entry + entry->length);
    }
    
    return 0; // No more entries
  }

  
  else if ( ip->type == T_FILE) {
    int len = size<16?size:16;
    if (offset < ip->size) {
      memmove(dst,"Very Fake Text.",len);
      return len;
    }
    else {
      return 0;
    }
  }
  return -1; // return -1 on error
}

struct inode_functions iso9660fs_functions = {
  iso9660fs_ipopulate,
  iso9660fs_iupdate,
  iso9660fs_readi,
  iso9660fs_writei,
};

    mount_point->i_func = &iso9660fs_functions;
    mount_point->mounted_dev = 2;
    mount_point->addrs[0]=pvd.root_directory_record.extent*2048; 
    mount_point->type = T_DIR;   
    mount_point->size = pvd.root_directory_record.size;
    iunlock(mount_point);
