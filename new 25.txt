import numpy as np
import random
from collections import defaultdict
import pyspiel
from open_spiel.python import rl_agent
from open_spiel.python.examples import gto_kuhn_poker

import matplotlib.pyplot as plt
import matplotlib.ticker as mticker


class RWYWEAgent(rl_agent.AbstractAgent):
    """
    Risk What You've Won in Expectation (RWYWE) agent for Kuhn poker.

    Based on Ganzfried & Sandholm (2015) "Safe Opponent Exploitation."

    Core invariant (Proposition 6.4 / Algorithm 2):
        At each iteration t, play a strategy π_t ∈ SAFE(k_t), then update:

            k_{t+1} = k_t + u_i(π_t_i, a^t_{-i}) - v*

        where u_i(π_t_i, a^t_{-i}) is our *expected* payoff of our *mixed*
        strategy against the opponent's *observed* action (not our realized
        payoff).  Using expected rather than realized payoff is exactly what
        separates RWYWE (safe) from RWYW (unsafe): it removes the variance
        due to our own randomization while retaining the signal from the
        opponent's actual play.

    SAFE(k) is the set of strategies with exploitability ≤ k.  We approximate
    it here by rollout-estimated EVs: we play a full best response (our
    highest-EV action) whenever doing so keeps exploitability within k,
    otherwise we fall back to the GTO strategy (exploitability = 0).

    Opponent model:
        We track action frequencies per information-state string for the
        opponent.  This gives us a distribution over the opponent's actions
        at each infoset, which we use to estimate EVs via rollouts biased
        toward the observed opponent distribution.
    """

    def __init__(self, game: pyspiel.Game, player_id: int, name='rwywe_agent'):
        super().__init__(game, player_id, name)
        self.game = game
        self.player_id = player_id
        self.opp_id = 1 - player_id

        # v* = value of Kuhn poker to player 1 = -1/18
        self.v_star = -1.0 / 18.0

        # k_t: cumulative expected-profit surplus above v* (Algorithm 2).
        # Starts at 0; stays ≥ 0 by Lemma 6.3.  Represents how much
        # exploitability budget we can "spend" on a non-GTO strategy.
        self.k = 0.0

        # Opponent model: infoset_string -> {action -> count}
        # Used to compute a frequency-based model of the opponent's strategy.
        self.opp_model = defaultdict(lambda: defaultdict(int))

        # GTO baseline (α = 0.2 as specified).  Exploitability = 0 by
        # definition, so it is always in SAFE(k) for any k ≥ 0.
        self.gto_policy = gto_kuhn_poker.GTOKuhnPolicy(game, alpha=0.2)  # GTOKuhnPolicy.action_probabilities() is correct for all infosets

        # Number of MC rollouts per action for EV estimation.
        self.num_rollouts = 100

        # Per-hand bookkeeping: stores (mixed_strategy_ev, opponent_action)
        # tuples so we can compute the RWYWE update at terminal time.
        # mixed_strategy_ev = u_i(π_t_i, a^t_{-i})  (the key RWYWE quantity)
        self._pending_mixed_ev = None   # set in step(), consumed in on_terminal()
        self._pending_action_probs = None  # mixed strategy used this hand

    # ------------------------------------------------------------------
    # Rollout / EV estimation helpers
    # ------------------------------------------------------------------

    def _simulate_rollout(self, state):
        """Play out a state uniformly at random; return our player's return."""
        s = state.clone()
        while not s.is_terminal():
            legal = s.legal_actions()
            a = legal[random.randint(0, len(legal) - 1)]
            s.apply_action(a)
        return s.returns()[self.player_id]

    def _opp_action_probs(self, info_str):
        """
        Return a probability distribution over opponent actions at *info_str*
        derived from observed frequencies.  Falls back to uniform if unseen.
        """
        counts = self.opp_model[info_str]
        if not counts:
            return None  # unseen infoset → caller should use uniform / GTO
        total = sum(counts.values())
        return {a: c / total for a, c in counts.items()}

    def _estimate_action_ev(self, state, action):
        """
        Estimate E[return | we play *action* now] via MC rollouts.

        For the purpose of computing our *mixed-strategy expected payoff*
        against the opponent's observed action (the RWYWE update term), we
        need EV estimates per action so we can take the weighted average
        under our chosen mixed strategy.
        """
        total = 0.0
        for _ in range(self.num_rollouts):
            s = state.clone()
            s.apply_action(action)
            total += self._simulate_rollout(s)
        return total / self.num_rollouts

    # ------------------------------------------------------------------
    # Core decision logic
    # ------------------------------------------------------------------

    def _exploitability_of_pure(self, state, action):
        """
        Conservative upper-bound on exploitability if we commit to *action*.

        A pure strategy that is not the GTO action could be exploited by the
        opponent's nemesis.  We estimate exploitability as:

            expl ≈ v* - min_{s_{-i}} u_i(action, s_{-i})

        i.e. how much worse than v* our worst-case payoff could be.  For
        Kuhn poker the worst-case opponent response to any fixed action is
        bounded; we approximate it by subtracting the worst observed rollout
        EV from v*.  A tighter bound would require a full LP, but this keeps
        the implementation self-contained and matches the paper's spirit of
        using approximate safe best responses.

        In practice: if our MC EV estimate for *action* ≥ v* - k, we treat
        it as being within SAFE(k).  This is conservative (we may
        under-exploit) but safe.
        """
        ev = self._estimate_action_ev(state, action)
        # exploitability proxy: how far below v* could we end up?
        # We approximate worst-case EV as ev - 1 std-dev safety margin.
        worst_case_ev = ev - np.sqrt(1.0 / self.num_rollouts)  # ≥ true min w.h.p.
        expl_estimate = max(0.0, self.v_star - worst_case_ev)
        return ev, expl_estimate

    def step(self, state):
        """
        Select an action using RWYWE (Algorithm 2 extended to extensive-form,
        Algorithm 5/6).

        Decision rule:
          1. Estimate EV and exploitability for each legal action.
          2. Find the best-EV action whose exploitability ≤ k  (ε-safe best
             response with ε = k).
          3. If no such action exists (k is too small to allow any deviation),
             fall back to the GTO strategy (exploitability = 0).

        After choosing the mixed strategy π_t (either a pure action or GTO),
        we record the probability we assign to each action so that
        on_terminal() can compute u_i(π_t_i, a^t_{-i}).
        """
        legal = state.legal_actions()

        # --- Estimate EV and exploitability for every legal action ---
        action_stats = {}  # action -> (ev, expl_estimate)
        for a in legal:
            ev, expl = self._exploitability_of_pure(state, a)
            action_stats[a] = (ev, expl)

        # --- Find the best-EV action within SAFE(k) ---
        # An action is "safe to play" if its exploitability estimate ≤ k.
        # This implements the ε-safe best response selection from RWYWE.
        safe_actions = [(a, ev) for a, (ev, expl) in action_stats.items()
                        if expl <= self.k]

        if safe_actions:
            # Play the highest-EV safe action (pure strategy within SAFE(k)).
            best_action, best_ev = max(safe_actions, key=lambda x: x[1])

            # Record the mixed strategy as a one-hot over the chosen action.
            # u_i(π_t_i, a^t_{-i}) will simply be the EV of best_action
            # against whatever action the opponent actually takes.
            # We store the per-action EVs so on_terminal() can look up
            # u_i(π_t_i, opponent_action) after we observe the opponent.
            self._pending_action_probs = {a: 1.0 if a == best_action else 0.0
                                          for a in legal}
            self._pending_action_evs = {a: action_stats[a][0] for a in legal}
            chosen_action = best_action
        else:
            # k is too small to deviate safely → play GTO (exploitability 0).
            # GTO is always in SAFE(k) for any k ≥ 0 (Lemma 6.3).
            # Use GTOKuhnPolicy.action_probabilities() — this is the correct,
            # complete implementation covering all infosets including "pb".
            gto_probs = self.gto_policy.action_probabilities(state)
            actions = list(gto_probs.keys())
            weights = [gto_probs[a] for a in actions]
            chosen_action = random.choices(actions, weights)[0]

            self._pending_action_probs = gto_probs
            self._pending_action_evs = {a: action_stats[a][0] for a in legal}

        return chosen_action

    # ------------------------------------------------------------------
    # Feedback hooks
    # ------------------------------------------------------------------

    def inform_action(self, state, player, action):
        """
        Called after the opponent acts.  We update our frequency-based
        opponent model and record the opponent's observed action so that
        on_terminal() can compute u_i(π_t_i, a^t_{-i}).
        """
        if player == self.opp_id:
            info = state.information_state_string(self.opp_id)
            self.opp_model[info][action] += 1

            # Store the opponent's action so on_terminal() can look it up.
            # In Kuhn poker there is exactly one opponent action per hand
            # (at most), so overwriting is fine.  In deeper games you would
            # need a list.
            self._last_opp_action = action

    def on_terminal(self, state):
        """
        Called at the end of each hand.  Implements the RWYWE k-update:

            k_{t+1} = k_t + u_i(π_t_i, a^t_{-i}) - v*

        where u_i(π_t_i, a^t_{-i}) is the *expected* payoff of our *mixed*
        strategy π_t against the opponent's *observed* action a^t_{-i}.

        This is the crucial difference from RWYW (which uses realized payoff):
        by taking the expectation over our own randomization we remove the
        noise from our own mixed strategy, making the update safe (Lemma 6.3,
        Proposition 6.4).

        If the opponent's action was not observed this hand (e.g., the hand
        ended before they acted), we conservatively use v* as the expected
        payoff, leaving k unchanged.
        """
        if self._pending_action_probs is None:
            # step() was never called this hand; nothing to update.
            return

        opp_action = getattr(self, '_last_opp_action', None)

        if opp_action is not None and self._pending_action_evs is not None:
            # u_i(π_t_i, a^t_{-i}) = Σ_a  π_t_i(a) · EV(a | opp played opp_action)
            # Because our EV estimates are computed against a random continuation
            # (not conditioned on the opponent's specific action), we use the
            # stored per-action EVs as a proxy.  A fully correct implementation
            # would condition the rollouts on the observed opponent action, but
            # for Kuhn poker (single betting round) this approximation is tight.
            mixed_ev = sum(
                self._pending_action_probs.get(a, 0.0) * ev
                for a, ev in self._pending_action_evs.items()
            )
        else:
            # Conservative fallback: assume we got exactly v* this hand.
            mixed_ev = self.v_star

        # RWYWE update (Algorithm 2, update step).
        # k stays ≥ 0 by Lemma 6.3; we clamp to enforce this numerically.
        self.k = max(0.0, self.k + mixed_ev - self.v_star)

        # Reset per-hand state.
        self._pending_action_probs = None
        self._pending_action_evs = None
        self._last_opp_action = None

    def restart(self):
        """Reset per-episode state (does NOT reset k or the opponent model)."""
        self._pending_action_probs = None
        self._pending_action_evs = None
        self._last_opp_action = None


def run_simulation(num_hands: int = 1000, seed: int = 42):
    random.seed(seed)
    np.random.seed(seed)

    game = pyspiel.load_game("kuhn_poker")
    agent = RWYWEAgent(game, player_id=0)
    gto = gto_kuhn_poker.GTOKuhnPoker(game, 1)
    # Simple random opponent
    def random_policy(state):
        return random.choice(state.legal_actions())

    k_history = []
    gain_history = []
    cumulative_gain = 0.0

    for _ in range(num_hands):
        state = game.new_initial_state()

        while not state.is_terminal():
            cur = state.current_player()
            if cur == pyspiel.PlayerId.CHANCE:
                outcomes, probs = zip(*state.chance_outcomes())
                state.apply_action(np.random.choice(outcomes, p=probs))
            elif cur == agent.player_id:
                action = agent.step(state)
                state.apply_action(action)
            else:
                opp_action = random_policy(state)
                agent.inform_action(state, cur, opp_action)
                state.apply_action(opp_action)

        agent.on_terminal(state)

        hand_payoff = state.returns()[agent.player_id]
        cumulative_gain += hand_payoff

        k_history.append(agent.k)
        gain_history.append(cumulative_gain)

    return k_history, gain_history


def plot_results(k_history, gain_history):
    hands = range(1, len(k_history) + 1)

    fig, ax1 = plt.subplots(figsize=(11, 5))
    fig.patch.set_facecolor("white")
    ax1.set_facecolor("white")

    # --- Cumulative gain (left axis) ---
    color_gain = "#16a34a"
    ax1.plot(hands, gain_history, color=color_gain, linewidth=1.4,
             alpha=0.85, label="Cumulative gain")
    ax1.axhline(0, color="#cccccc", linewidth=0.8, linestyle="--")
    ax1.set_xlabel("Hand", color="black", fontsize=11)
    ax1.set_ylabel("Cumulative gain (chips)", color=color_gain, fontsize=11)
    ax1.tick_params(axis="y", colors=color_gain)
    ax1.tick_params(axis="x", colors="black")
    for spine in ax1.spines.values():
        spine.set_edgecolor("#cccccc")

    # --- k budget (right axis) ---
    ax2 = ax1.twinx()
    ax2.set_facecolor("white")
    color_k = "#c2410c"
    ax2.plot(hands, k_history, color=color_k, linewidth=1.4,
             alpha=0.75, linestyle="--", label="k (exploit budget)")
    ax2.set_ylabel("k  (exploit budget)", color=color_k, fontsize=11)
    ax2.tick_params(axis="y", colors=color_k)
    for spine in ax2.spines.values():
        spine.set_edgecolor("#cccccc")

    # --- Legend ---
    lines1, labels1 = ax1.get_legend_handles_labels()
    lines2, labels2 = ax2.get_legend_handles_labels()
    legend = ax1.legend(lines1 + lines2, labels1 + labels2,
                        loc="upper left", framealpha=0.25,
                        facecolor="white", edgecolor="#cccccc",
                        labelcolor="black", fontsize=10)

    fig.suptitle("RWYWE Agent — 1 000 hands of Kuhn Poker\nvs. random opponent",
                 color="black", fontsize=13, y=0.98)
    fig.tight_layout()
    plt.savefig("rwywe_simulation.png", dpi=150, bbox_inches="tight",
                facecolor="white")
    plt.show()
    print("Chart saved to rwywe_simulation.png")


if __name__ == "__main__":
    k_hist, gain_hist = run_simulation(num_hands=1000)
    plot_results(k_hist, gain_hist)
